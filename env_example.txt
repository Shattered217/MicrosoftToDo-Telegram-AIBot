# 环境配置模板
# 复制此文件为 .env 并填入你的实际配置

# ==================== Telegram Bot ====================
# Telegram Bot Token，从 @BotFather 获取
TELEGRAM_BOT_TOKEN=your_telegram_bot_token_here

# Telegram 管理员用户 ID（多个用逗号分隔）
TELEGRAM_ADMIN_IDS=your_telegram_user_id

# 可选：自定义 Telegram Bot API Base URL（例如通过 EdgeOne/反向代理加速）
#TELEGRAM_BASE_URL=https://your-edgeone-or-proxy.example.com/bot

# ==================== OpenAI API ====================
# OpenAI 兼容 API 密钥
OPENAI_API_KEY=your_openai_api_key_here

# API 基础 URL
# OpenAI 官方: https://api.openai.com/v1
# SiliconFlow（推荐，国内访问快）: https://api.siliconflow.cn/v1
OPENAI_BASE_URL=https://api.siliconflow.cn/v1

# 文本模型，用于任务分析、匹配和分解
# 推荐模型: Qwen/Qwen3-30B-A3B-Instruct-2507
OPENAI_MODEL=Qwen/Qwen3-30B-A3B-Instruct-2507

# Vision 模型，用于图片分析
# 推荐模型: Qwen/Qwen3-Omni-30B-A3B-Captioner
OPENAI_VL_MODEL=Qwen/Qwen3-Omni-30B-A3B-Captioner

# ==================== Microsoft To Do ====================
# 在 https://portal.azure.com 注册应用获取
MS_TODO_CLIENT_ID=your_azure_app_client_id

# 客户端密钥（可选，仅工作/学校账户支持）
#MS_TODO_CLIENT_SECRET=your_client_secret_here
#需要工作/学校账户才可使用密钥实现自动获取refresh_token，填入密钥即视为工作/学校账户

# 访问令牌（会自动刷新）
MS_TODO_ACCESS_TOKEN=your_access_token

# 刷新令牌（会自动刷新）
MS_TODO_REFRESH_TOKEN=your_refresh_token

# ==================== ESP32 Server（可选）====================
# ESP32 设备访问 API 的密钥
ESP32_API_KEY=your_esp32_api_key

# 服务器绑定地址（0.0.0.0 表示所有接口，127.0.0.1 表示仅本地）
ESP32_SERVER_HOST=0.0.0.0

# 服务器端口
ESP32_SERVER_PORT=5000

# ==================== 时区 ====================
# 用于提醒/截止时间的本地化，默认 Asia/Shanghai
